@app.route("/change_password", methods=["GET", "POST"])
@login_required
def change_password():
    """Personal touch to app, decided on add funds, change password and buy/sell on portfolio page"""
    """Enable user to change password"""
    if request.method == "GET":
        return render_template("account.html")
    else:
        user_id = session["user_id"]

        # Check that current password was entered
        if not request.form.get("old"):
            return apology("Enter your current password")

        # Retrive current user password and ensure it matches
        user_password = db.execute("SELECT hash FROM users WHERE id = ?", user_id)[0]["hash"]
        if not check_password_hash(user_password, request.form.get("old")):
            return apology("Invalid Password")

        # Ensure new password was entered
        if not request.form.get("new"):
            return apology("Enter new password")

        # Ensure confirmation password was entered
        elif not request.form.get("confirmation"):
            return apology("Confirm new password")

        # Ensure new and confirmation passwords match
        elif request.form.get("new") != request.form.get("confirmation"):
            return apology("New Password and confirmation password do not match")

        # Hash the new password and update the database
        new_password = generate_password_hash(request.form.get("new"))
        db.execute("UPDATE users SET hash = ? WHERE id = ?", new_password, user_id)

        # Flash success message
        flash("Password has been Changed")

        return redirect("/")
